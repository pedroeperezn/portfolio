using System;
class GalacticAdventurer
{
  static void Main()
  {
      Console.WriteLine("Welcome, Galactic Adventurer!");

      // Generate a random space creature name
      string spaceCreature = GenerateRandomSpaceCreature();
      Console.WriteLine($"You are about to encounter a fearsome {spaceCreature}!");

      // Simulate a space battle
      Console.WriteLine("Initiating Space Battle...");

      int playerHealth = 100;
      int spaceCreatureHealth = new Random().Next(50, 101);

      while (playerHealth > 0 && spaceCreatureHealth > 0)
      {
          // Player attacks the space creature
          int playerAttack = new Random().Next(10, 21);
          spaceCreatureHealth -= playerAttack;
          Console.WriteLine($"You attack the {spaceCreature} and deal {playerAttack} damage!");

          // Space creature counterattacks
          int creatureAttack = new Random().Next(5, 16);
          playerHealth -= creatureAttack;
          Console.WriteLine($"The {spaceCreature} counterattacks and deals {creatureAttack} damage!");

          // Display health status
          Console.WriteLine($"Your Health: {playerHealth} | {spaceCreature} Health: {spaceCreatureHealth}");
          Console.WriteLine();
      }

      // Display the outcome of the space battle
      if (playerHealth > 0)
      {
          Console.WriteLine($"Congratulations! You defeated the {spaceCreature}. You are a true Galactic Hero!");
      }
      else
      {
          Console.WriteLine($"Oh no! The {spaceCreature} was too powerful. Better luck next time, Galactic Adventurer.");
      }
  }

  static string GenerateRandomSpaceCreature()
  {
      string[] spaceCreatures = { "Zorgon", "Nebula Serpent", "Quasar Beast", "Galactic Kraken", "Stellar Phoenix" };
      int randomIndex = new Random().Next(spaceCreatures.Length);
      return spaceCreatures[randomIndex];
  }
}

using System;
class CyberPetSimulator
{
    static void Main()
    {
        Console.WriteLine("Welcome to the Cyber Pet Simulator!");

        // Create a virtual cyber pet
        CyberPet myPet = new CyberPet("ByteBot");

        // Interact with the cyber pet
        Console.WriteLine($"You adopted a cyber pet named {myPet.Name}!");
        Console.WriteLine($"Initial happiness level: {myPet.Happiness}");

        Console.WriteLine("Playing with your cyber pet...");
        myPet.Play();
        Console.WriteLine($"Happiness after playtime: {myPet.Happiness}");

        Console.WriteLine("Feeding your cyber pet...");
        myPet.Feed();
        Console.WriteLine($"Happiness after feeding: {myPet.Happiness}");

        Console.WriteLine("Taking your cyber pet to the virtual doctor...");
        myPet.VisitDoctor();
        Console.WriteLine($"Happiness after the virtual doctor visit: {myPet.Happiness}");

        Console.WriteLine("Your cyber pet says goodbye!");
    }
}

class CyberPet
{
    public string Name { get; }
    public int Happiness { get; private set; }

    public CyberPet(string name)
    {
        Name = name;
        Happiness = 50; // Initial happiness level
    }

    public void Play()
    {
        Console.WriteLine($"Playing with {Name}...");
        Happiness += 20;
    }

    public void Feed()
    {
        Console.WriteLine($"Feeding {Name}...");
        Happiness += 10;
    }

    public void VisitDoctor()
    {
        Console.WriteLine($"Taking {Name} to the virtual doctor...");
        Happiness -= 15;
    }
}

{

    {
